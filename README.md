#include <iostream>
#include <vector>
#include <thread>
#include <pcap.h>
#include "anomaly_detector.h"

class PacketAnalyzer {
private:
    pcap_t* handle;
    AnomalyDetector detector;
    std::vector<NetworkFlow> active_flows;
    
public:
    PacketAnalyzer(const char* device) {
        char errbuf[PCAP_ERRBUF_SIZE];
        
        // Open device for live capture
        handle = pcap_open_live(device, BUFSIZ, 1, 1000, errbuf);
        if (handle == NULL) {
            std::cerr << "Could not open device " << device << ": " << errbuf << std::endl;
            exit(1);
        }
        
        std::cout << "âœ… Initialized packet capture on " << device << std::endl;
        detector.load_baseline_model("./models/baseline.ml");
    }
    
    void analyze_packet(const struct pcap_pkthdr* header, const u_char* packet) {
        // Extract flow information
        NetworkFlow flow = extract_flow_data(packet, header->len);
        
        // Deep packet inspection
        if (detector.is_anomalous(flow)) {
            alert("âš ï¸ Suspicious traffic detected!", flow);
            log_event(flow, "anomaly");
        }
        
        // Update traffic statistics
        update_dashboard(flow);
    }
    
    void start_capture() {
        std::cout << "ğŸ” Starting packet analysis..." << std::endl;
        pcap_loop(handle, 0, packet_handler, reinterpret_cast<u_char*>(this));
    }
};

int main(int argc, char* argv[]) {
    if (argc != 2) {
        std::cout << "Usage: " << argv[0] << " <interface>" << std::endl;
        return 1;
    }
    
    PacketAnalyzer analyzer(argv[1]);
    analyzer.start_capture();
    
    return 0;
}
